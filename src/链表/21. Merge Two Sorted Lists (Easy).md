[21. 合并两个有序链表](https://leetcode-cn.com/problems/merge-two-sorted-lists/description/)



# 我的解法：转为数组排序
- 把两个链表放入数组，排序后返回链表
- 还挺快的

```C++
class Solution {
public:
    vector<int> twoLists;
    void traversalList(ListNode* node)
    {
        while(node!=NULL) 
        {
            twoLists.push_back(node->val);
            node=node->next;
        }
    }
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        traversalList(l1);
        traversalList(l2);
        sort(twoLists.begin(),twoLists.end());
        ListNode *head = NULL;
        ListNode *cur = NULL;
        for(int i=0;i<twoLists.size();i++)
        {
            if(head==NULL){
                head = new ListNode(twoLists[i]);
                cur=head;
            }
            else{
                cur->next=new ListNode(twoLists[i]);
                cur=cur->next;
            }
        }

        return head;
    }
};
```

# 解法一：递归
- 哈希表也可以存放链表
```C++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if (l1 == nullptr) {
            return l2;
        } else if (l2 == nullptr) {
            return l1;
        } else if (l1->val < l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        } else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }
};

```

# 解法二：迭代
```c++

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* preHead = new ListNode(-1);

        ListNode* prev = preHead;
        while (l1 != nullptr && l2 != nullptr) {
            if (l1->val < l2->val) {
                prev->next = l1;
                l1 = l1->next;
            } else {
                prev->next = l2;
                l2 = l2->next;
            }
            prev = prev->next;
        }

        // 合并后 l1 和 l2 最多只有一个还未被合并完，我们直接将链表末尾指向未合并完的链表即可
        prev->next = l1 == nullptr ? l2 : l1;

        return preHead->next;
    }
};


```

# 自己写的迭代
- 答题思想跟上面的迭代差不多，自己在写代码时候，对于构建链表处理的不好，输出链表长度一直为2，最后翻了链表建立的代码才最终改好

```c++

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
    ListNode *head = NULL;
    ListNode *cur = NULL;
    int nowMax=-100;
    if(l2==NULL) return l1;
    if(l1==NULL) return l2;
    while(l1!=NULL&&l2!=NULL)
    {
        
            if(l1->val-nowMax <= l2->val-nowMax) //取l1中数据
            {
                if(head==NULL) 
                {
                    head=new ListNode(l1->val);
                    nowMax=l1->val;
                    l1 = l1->next;
                    cur=head;
                    
                }
                else{
                    cur->next=new ListNode(l1->val);
                    cur=cur->next;
                    nowMax=l1->val;
                    l1 = l1->next;
                }
                
            }
            else{
                if(head==NULL) 
                {
                    head=new ListNode(l2->val);
                    nowMax=l2->val;
                    l2=l2->next;
                    cur=head;
                }
                else{
                    cur->next=new ListNode(l2->val);
                    cur=cur->next;
                    nowMax=l2->val;
                    l2=l2->next;
                
                }
            }
        }
    if(l2!=NULL) cur->next = l2;
    if(l1!=NULL) cur->next = l1;

        return head;
    }
};


```
