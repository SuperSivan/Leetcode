# 我的解法：先将所有元素遍历放到StoreNums一维数组，然后用for循环的形式以此放到新的二维数组中
```C++
    class Solution {
    public:
        void traversal(vector<vector<int>> vec, vector<int>& dst)
        {
            int i, j;
            //cout << "Use index : " << endl;
            for (i = 0; i < vec.size(); i++)
            {
                for (j = 0; j < vec[0].size(); j++)
                    dst.push_back(vec[i][j]);
            }
        }
        vector<vector<int>> matrixReshape(vector<vector<int>>& nums, int r, int c) {
            int cols = nums.size();
            int rows = nums[0].size();
            vector<int> StoreNums;
            vector<vector<int>> dst;

            if (cols*rows!=r*c)
                return nums;
            else
            {
                traversal(nums, StoreNums);

                for (int i = 0; i < r ; i++)
                {
                    vector<int> temp;
                    for (int j = 0; j < c ; j++)
                    {
                        temp.push_back(StoreNums[i*c+j]);
                    }
                    dst.push_back(temp);
                }             
                    return dst;
            }

        }
    };
```

# 解法一:不需要通过一维数组进行过渡，直接从二维数组中求得对应映射关系
```c++
    class Solution {
    public:
        vector<vector<int>> matrixReshape(vector<vector<int>>& nums, int r, int c) {
            int m = nums.size();
            int n = nums[0].size();
            if (m * n != r * c) {
                return nums;
            }

            vector<vector<int>> ans(r, vector<int>(c));
            for (int x = 0; x < m * n; ++x) {
               ans[x / c][x % c] = nums[x / n][x % n];
            }
            return ans;
        }
    };


```
