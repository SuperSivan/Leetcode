[88. 合并两个有序数组](https://leetcode-cn.com/problems/merge-sorted-array/description/)



# 解法一：无脑排序
```C++
class Solution {
public:

    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        if(n==0) return;
        for(int i=0;i<nums2.size();i++)
        {
            nums1[m+i]=nums2[i];
        }
        sort(nums1.begin(),nums1.end());
    }
};
```
