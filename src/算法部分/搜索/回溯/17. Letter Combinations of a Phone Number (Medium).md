[17. 电话号码的字母组合](https://leetcode-cn.com/problems/letter-combinations-of-a-phone-number/description/)




# 我的解法：BFS
- 注意char到string的转换
```C++
class Solution {
public:
    vector<string> letterCombinations(string digits) {
        vector<string> words={"a","a","abc", "def", "ghi","jkl","mno","pqrs","tuv","wxyz"};
        queue<string> q;
        vector<string> ans;
        int n = digits.size();
        if(n==0) return ans;


        //先把第一个放进去
        for(int i=0;i<words[digits[0]-'0'].size();i++)
        {
            string str_temp={'1'};
            str_temp[0]=words[digits[0]-'0'][i];
            q.push(str_temp);
        }
        
        //BFS
        //不是while()!q.empty()开头了
        for(int k =1;k<n;k++)
        {
            int size = q.size();
            for(int i=0;i<size;i++)
            {
                string father = q.front();
                q.pop();
                for(int j=0;j<words[digits[k]-'0'].size();j++)
                {
                    string str_temp={'1'};
                    str_temp[0]=words[digits[k]-'0'][j];
                    q.push(father+str_temp);
                }
            }
        }
        //将q中的放入数组中
        while(!q.empty())
        {
            ans.push_back(q.front());
            q.pop();
        }
        return ans;
    }
};


```

class Solution {
public:
    string tmp={""};
    vector<string> res;
    vector<string> board={"","","abc","def","ghi","jkl","mno","pqrs","tuv","wxyz"};
    void dfs(int pos,string digits)
    {
        if(pos==digits.size()){
            res.push_back(tmp);
            return;
        }
        int num=digits[pos]-'0';
        
        string letters = board[num];
        for(int i=0;i<letters.size();i++){
            tmp.push_back(letters[i]);
            dfs(pos+1,digits);
            tmp.pop_back();
        }
    }
    vector<string> letterCombinations(string digits) {
        if(digits.size()==0) return res;
        
        dfs(0,digits);
        return res;
    }
};
