[69. x 的平方根](https://leetcode-cn.com/problems/sqrtx/description/)



# 解法一：二分
- 自己没搞出来
```C++
class Solution {
public:
    int mySqrt(int x) {
        int l=0,r=x,mid=0,ans=0;
        while(l<=r)
        {
            mid=(l+r)/2;
            if((long long) mid*mid>x)
            {
                r=mid-1;
            }
            else if(mid*mid==x)
            return mid;
            else
            {
                ans=mid;//这个是怎么想出来的
                l=mid+1;
            }
        }
        return ans;
    }
};
```
