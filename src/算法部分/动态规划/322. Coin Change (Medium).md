# 我的解法：动态规划（超时）
- 二维DP，超时
```C++
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int N=coins.size();
        if(N==1&&amount%coins[0]!=0) return -1;
        if(N==1&&amount%coins[0]==0) return amount/coins[0];
        if(amount==0) return 0;

        vector<vector<int>> dp(N,vector<int>(amount+1,-1));


        for(int i=0;i<N;i++)
            dp[i][0]=-1;

        for(int i=1;i<N;i++)
        {
            for(int j=1;j<=amount;j++)
            {
                if(j-coins[i]<0)
                    dp[i][j]=dp[i-1][j];
                else if(j-coins[i]==0)
                    dp[i][j]=1;
                else
                {
                    int minNum=INT_MAX;
                    for(int u=0;u<=i;u++)
                    {
                        if(dp[i][j-coins[u]]!=-1)
                            minNum=(minNum,dp[i][j-coins[u]]+1);
                        
                    }
                    if(minNum!=INT_MAX) dp[i][j]=minNum;
                }

            }
        }


        return dp[N-1][amount];

    }
};
```

```c++
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int N=coins.size();
        if(N==1&&amount%coins[0]!=0) return -1;
        if(N==1&&amount%coins[0]==0) return amount/coins[0];
        if(amount==0) return 0;
        //sort(coins.begin(),coins.end());
        //reverse(coins.begin(),coins.end());

        vector<int> dp(amount+1,-1);
        dp[0]=0;
        // for(int i=0;i<N;i++)//不能这样初始化，可能出现coins[i]>amount，导致dp[coins[i]]越界
        //     dp[coins[i]]=1;
        for(int i=1;i<=amount;i++)
        {
            int minNum = INT_MAX;
           for (int j = 0; j < (int)coins.size(); j++) 
           {
                if (coins[j] <= i&&dp[i - coins[j]]!=-1) {
                    minNum = min(minNum, dp[i - coins[j]] + 1);
                }
            }
            if(minNum!=INT_MAX) dp[i]=minNum;
        }
        return dp[amount];

    }
};
```
