[547. 省份数量](https://leetcode-cn.com/problems/friend-circles/description//)

# 我的解法:DFS
```c++
class Solution {
public:
    void changeCol(int i,int index,vector<vector<int>>& isConnected,vector<vector<int>>& visited)
    {//找到一个点为1如[1,3]，则下一步检查第三行，检测到的点都属于这个index
        for(int j=0;j<isConnected[0].size();j++)
        {
            if(isConnected[i][j]==1 && visited[i][j] ==0)
            {
                visited[i][j] = 1;
                changeCol(j,index,isConnected,visited);
            }
        }
    }
    int findCircleNum(vector<vector<int>>& isConnected) {
        int n = isConnected.size();
        if(n==0) return 0;
        int m = isConnected[0].size();
        int index = 1;
        vector<vector<int>> visited(n,vector<int>(m,0));

        for(int i = 0;i < n;i++)
            for(int j = 0;j < m;j++)
            {
                if(isConnected[i][j]==1 && visited[i][j] ==0)
                {
                    changeCol(i,index,isConnected,visited);
                    index++;
                }
            }

        return index-1;
    }
};

```
# 解法一：DFS(将二维简化到一维)

```C++
class Solution {
public:
    int n ;
    void dfs(vector<vector<int>>& isConnected,vector<bool>& visited,int i)
    {
        visited[i]= true;
        for (int j=0;j<isConnected.size();j++) 
        {
            if (isConnected[i][j] == 1 && !visited[j]) 
            {
                dfs(isConnected, visited, j);
            }
        }
    }
    int findCircleNum(vector<vector<int>>& isConnected) {
        n = isConnected.size();
        vector<bool> visited(n,0);
        int count = 0;
        for (int i=0;i<n;i++) 
        { // 遍历所有的城市

            if (visited[i]==0) //未访问
            {
                dfs(isConnected, visited, i);
                count++;
            }
        }
        return count;

    }
};

```

# 解法二：并查集
```c++
class Solution {
public:
    int find(vector<int>& parent,int index)
    {
        if (parent[index] != index) 
        {
            parent[index] = find(parent, parent[index]);
        }
        return parent[index];
    }
    void Union(vector<int>& parent, int index1, int index2) {
        parent[find(parent, index1)] = find(parent, index2);
    }
    int findCircleNum(vector<vector<int>>& isConnected) {
        int  n = isConnected.size();
        vector<int> parent(n,0);
        for(int i = 0;i < n;i++)
            parent[i] = i;

        for(int i = 0;i < n;i++)
            for(int j = i+1;j < n;j++)
            {
                if(isConnected[i][j]==1)
                    Union(parent,i,j);
            }
        int circles = 0;
        for (int i = 0; i < n; i++) {
            if (parent[i] == i) {
                circles++;
            }
        }
        return circles;
    }
};
```
